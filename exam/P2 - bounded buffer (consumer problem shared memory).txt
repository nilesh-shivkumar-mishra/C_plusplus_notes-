#include <iostream>
#include <cstdlib>

// Global variables to simulate semaphore values
int mutex = 1;
int full = 0;
int empty = 3;
int x = 0;

// Function prototypes
void producer();
void consumer();
int wait(int s);
int signal(int s);

int main()
{
    int n;
    std::cout << "\n1. Producer\n2. Consumer\n3. Exit";

    while (true)
    {
        std::cout << "\nEnter your choice: ";
        std::cin >> n;

        switch (n)
        {
            case 1:
                if (mutex == 1 && empty != 0)
                    producer();
                else
                    std::cout << "Buffer is full\n";
                break;

            case 2:
                if (mutex == 1 && full != 0)
                    consumer();
                else
                    std::cout << "\nBuffer is empty\n";
                break;

            case 3:
                std::cout << "Exiting program.\n";
                return 0; // Graceful termination

            default:
                std::cout << "Invalid choice\n";
        }
    }

    return 0;
}

int wait(int s)
{
    return (--s);
}

int signal(int s)
{
    return (++s);
}

void producer()
{
    mutex = wait(mutex);
    full = signal(full);
    empty = wait(empty);
    x++;
    std::cout << "\nProducer produces item " << x << "\n";
    mutex = signal(mutex);
}

void consumer()
{
    mutex = wait(mutex);
    full = wait(full);
    empty = signal(empty);
    std::cout << "\nConsumer consumes item " << x << "\n";
    x--;
    mutex = signal(mutex);
}
